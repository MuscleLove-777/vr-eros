{{/* BlogPosting JSON-LD ------------------- Emits a schema.org BlogPosting structured-data block for single pages. Coexists with the Article / Review / VideoObject / BreadcrumbList / WebSite JSON-LD blocks emitted from partials/extend_head.html, and with the PaperMod theme's built-in schema block. Rendered only on .Kind == "page" (regular article pages). Safe to invoke from any template — non-page kinds no-op. */}} {{- if eq .Kind "page" -}} {{- $desc := "" -}} {{- with .Description -}} {{- $desc = . -}} {{- else -}} {{- $desc = .Summary | plainify -}} {{- end -}} {{/* description が短すぎ (50 文字未満) ならリッチリザルト品質を担保するため Summary 流用で 80～160 文字の本文に拡張する。Rich Results 監査 2026-04-25 で 12 URL 中 3 件が 48-49 文字判定。これを自動回避。 */}} {{- if lt (len $desc) 50 -}} {{- $supplement := .Summary | plainify -}} {{- if and $supplement (ne $supplement $desc) -}} {{- $desc = printf "%s %s" $desc $supplement -}} {{- end -}} {{- end -}} {{- if gt (len $desc) 160 -}} {{- $desc = substr $desc 0 160 -}} {{- end -}} {{- $img := "" -}} {{- with .Params.cover -}}{{- with .image -}}{{- $img = . -}}{{- end -}}{{- end -}} {{- $absImg := "" -}} {{- if $img -}} {{- if or (hasPrefix $img "http://") (hasPrefix $img "https://") -}} {{- $absImg = $img -}} {{- else -}} {{- $absImg = printf "%s%s" (.Site.BaseURL | strings.TrimRight "/") (printf "/%s" ($img | strings.TrimLeft "/")) -}} {{- end -}} {{- end -}} {{- $authorName := .Site.Params.Author | default "エロナビ編集部" -}} {{- $section := "" -}} {{- with .Params.categories -}}{{- $section = index . 0 -}}{{- end -}} {{- end -}}
